🔧 Advanced Circuit Breaker with Metrics

import time
import functools
from prometheus_client import Counter, Histogram

class AdvancedCircuitBreaker:
    def __init__(self, failure_threshold=5, timeout=60, expected_exception=Exception):
        self.failure_threshold = failure_threshold
        self.timeout = timeout
        self.expected_exception = expected_exception
        self.failure_count = 0
        self.last_failure_time = None
        self.state = 'CLOSED'
        
        # Prometheus metrics
        self.circuit_state = Gauge('circuit_breaker_state', 'Circuit breaker state')
        self.failure_counter = Counter('circuit_breaker_failures', 'Circuit breaker failures')
        self.success_counter = Counter('circuit_breaker_successes', 'Circuit breaker successes')
        self.request_duration = Histogram('circuit_breaker_request_duration', 'Request duration')

    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            start_time = time.time()
            
            if self.state == 'OPEN':
                if time.time() - self.last_failure_time > self.timeout:
                    self.state = 'HALF_OPEN'
                    self.circuit_state.set(1)  # HALF_OPEN
                else:
                    self.failure_counter.inc()
                    raise Exception("Circuit breaker is OPEN")
            
            try:
                result = func(*args, **kwargs)
                self._on_success()
                self.request_duration.observe(time.time() - start_time)
                return result
            except self.expected_exception as e:
                self._on_failure()
                self.request_duration.observe(time.time() - start_time)
                raise e
        
        return wrapper

🚀 Distributed Rate Limiter with Redis Cluster

import redis
import time
import json
from typing import Dict, Optional

class ClusteredRateLimiter:
    def __init__(self, redis_cluster, key_prefix: str, max_requests: int, time_window: int):
        self.redis = redis_cluster
        self.key_prefix = key_prefix
        self.max_requests = max_requests
        self.time_window = time_window
        
    def allow_request(self, identifier: str, custom_window: Optional[int] = None) -> Dict:
        """Check if request is allowed and return detailed info"""
        key = f"{self.key_prefix}:{identifier}"
        current_time = time.time()
        window = custom_window or self.time_window
        
        # Lua script for atomic operations
        lua_script = """
        local key = KEYS[1]
        local current_time = tonumber(ARGV[1])
        local window = tonumber(ARGV[2])
        local max_requests = tonumber(ARGV[3])
        
        -- Remove old entries
        redis.call('zremrangebyscore', key, 0, current_time - window)
        
        -- Get current count
        local current_count = redis.call('zcard', key)
        
        -- Check if under limit
        if current_count < max_requests then
            redis.call('zadd', key, current_time, current_time)
            redis.call('expire', key, window)
            return {1, current_count + 1, max_requests - current_count - 1}
        else
            local oldest_request = redis.call('zrange', key, 0, 0, 'withscores')
            return {0, current_count, 0, oldest_request[1]}
        end
        """
        
        result = self.redis.eval(lua_script, 1, key, current_time, window, self.max_requests)
        
        return {
            'allowed': bool(result[0]),
            'current_count': result[1],
            'remaining': result[2],
            'reset_time': result[3] if len(result) > 3 else None
        }
